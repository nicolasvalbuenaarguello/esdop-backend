create view view_resultados as select *  ,
	 CASE WHEN res_accion like '%CAPTURADO%' THEN 'CAPTURAS'
	 	  WHEN res_accion like 'SOMETIMIENTO A LA JUSTICIA' THEN 'SOMETIMIENTO A LA JUSTICIA'
		  WHEN res_accion like '%ENTREGA%' THEN 'ENTREGAS'
		  WHEN res_accion like '%PRESENTACIONES VOLUNTARIA%' THEN 'PRESENTACIONES VOLUNTARIA'
		  WHEN res_accion like '%MUERTOS EN DESARROLLO DE OPERACIONES MILITARES%' THEN 'MDOM'
		  WHEN res_accion like '%ASESINADO(A)%' THEN 'ASESINADO(A)'
		  WHEN res_accion like '%HERIDO(A)%' THEN 'HERIDO(A)'
		  WHEN res_accion like '%LIBERADO%' THEN 'LIBERADOS'
		  WHEN res_accion like '%RESCATADO(A)%' THEN 'RESCATADOS'
            ELSE '---'
       END as "RESULTADOS",
	   	CASE WHEN res_accion like '%RME%' THEN 'MENORES RECUPERADOS'
            ELSE '---'
       END
	   as "MENORES"
from resultados;

	 CASE WHEN HOP_HECHO_POS like 'SI' THEN  1
	 	  WHEN HOP_HECHO_POS like 'NO' THEN -1
		  ELSE -1
		  
           
       END as "VALOR HECHO",

	 	  WHEN HOP_HECHO_POS like 'NO' THEN 'NEGATIVO'

       END as "TIPO HECHO"


from hechos;

-- ===========================================
-- Tabla 1: valor_material
-- Guarda materiales y su valor en COP
-- ===========================================

CREATE TABLE valor_material (
    id SERIAL PRIMARY KEY,                -- Identificador único
    variable TEXT NOT NULL,       -- Nombre del material (ej: "Acero")
    valor_cop NUMERIC(18,2) NOT NULL,     -- Valor en pesos colombianos
    fecha_registro TIMESTAMP DEFAULT NOW()  -- Fecha automática de registro
);

-- ===========================================
-- Tabla 2: valor_dolar
-- Guarda valores del dólar con su fecha
-- ===========================================

CREATE TABLE valor_dolar (
    id SERIAL PRIMARY KEY,                -- Identificador único
    valor_usd NUMERIC(18,2) NOT NULL,     -- Valor del dólar (ej: 4100.50)
    fecha_cotizacion DATE NOT NULL,       -- Fecha del valor del dólar
    fecha_registro TIMESTAMP DEFAULT NOW()  -- Fecha automática de registro
);

-- ===========================================
-- Índices recomendados
-- ===========================================

CREATE INDEX idx_valor_variable_variable ON valor_material(variable);
CREATE INDEX idx_valor_dolar_fecha ON valor_dolar(fecha_cotizacion);
